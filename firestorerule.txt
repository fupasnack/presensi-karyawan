// firestore.rules
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Admin dapat membaca dan menulis semua dokumen
    match /{document=**} {
      allow read, write: if request.auth != null && 
        (request.auth.token.admin == true || 
         request.auth.uid in ['DsBQ1TdWjgXvpVHUQJpF1H6jZzJ3', 'xxySAjSMqKeq7SC6r5vyzes7USY2']);
    }
    
    // Karyawan hanya dapat membaca dan menulis data mereka sendiri
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    match /presensi/{presensiId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && 
        (request.auth.token.admin == true || 
         request.auth.uid == resource.data.uid);
    }
    
    match /cuti/{cutiId} {
      allow read: if request.auth != null && 
        (request.auth.token.admin == true || 
         request.auth.uid == resource.data.uid);
      allow write: if request.auth != null && 
        (request.auth.token.admin == true || 
         request.auth.uid == resource.data.uid);
    }
    
    match /notifikasi/{notifId} {
      allow read, write: if request.auth != null && 
        (request.auth.token.admin == true || 
         resource.data.untuk == request.auth.uid || 
         resource.data.untuk == 'all');
    }
  }
}
